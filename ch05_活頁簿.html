127 參照活頁簿¶

Attribute VB_Name = "Module1"
Option Explicit

Sub 參照活頁簿()
    Dim xBook As Variant, i As Integer
    xBook = Array("課題1.xls", "課題2.xls", "課題3.xls")
    For i = 0 To 2
        Workbooks.Open Filename:=xBook(i)
    Next
    MsgBox Workbooks(1).Name & ":" & Workbooks(2).Name & ":" & _
        Workbooks(3).Name & ":" & Workbooks(4).Name
End Sub

129 將活頁簿轉為作用中¶

Attribute VB_Name = "Module1"
Option Explicit

Sub 作用活頁簿()
    Dim xBook As Variant, i As Integer
    xBook = Array("課題1.xls", "課題2.xls", "課題3.xls")
    For i = 0 To 2
        Workbooks.Open Filename:=xBook(i)
        Workbooks(1).Worksheets(1).Cells(i + 2, "B").Value = xBook(i)
    Next
    Workbooks(1).Activate
End Sub

136 檢查活頁簿是否有巨集再存檔¶

Attribute VB_Name = "Module1"
Option Explicit

Sub 檢查是否有巨集再存檔()
    Dim fPath As String, fName As String
    fPath = "C:\ExcelVBA\"
    fName = Format(Date, "mm_dd") & "結果"
    If ActiveWorkbook.HasVBProject Then
       ActiveWorkbook.SaveAs _
           Filename:=fPath & fName, _
           FileFormat:=xlOpenXMLWorkbookMacroEnabled
    Else
       ActiveWorkbook.SaveAs _
           Filename:=fPath & fName, _
           FileFormat:=xlOpenXMLWorkbook
    End If
End Sub

137 檢查活頁簿的修改是否已存檔¶

141 取得活頁簿的存檔路徑¶

Attribute VB_Name = "Module1"
Option Explicit

Sub 活頁簿的存檔地點()
    Dim fPath As String
    On Error GoTo errHandler
    
    fPath = ActiveWorkbook.Path
    ActiveSheet.Copy
    ActiveWorkbook.SaveAs Filename:=fPath & "\" & Range("A1").Value & "xls"
    MsgBox ActiveWorkbook.Path
    Exit Sub
errHandler:
    MsgBox Err.Description
End Sub

Sub 活頁簿的存檔地點2()
    On Error GoTo errHandler
     
    ChDir ActiveWorkbook.Path
    ActiveSheet.Copy
    ActiveWorkbook.SaveAs _
        Filename:=Range("A1").Value & ".xls"
    MsgBox ActiveWorkbook.Path
    Exit Sub
errHandler:
    MsgBox Err.Description
End Sub

144 開啟活頁簿時自動執行的程序¶

VERSION 1.0 CLASS
BEGIN
  MultiUse = -1  'True
END
Attribute VB_Name = "ThisWorkbook"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = True
Option Explicit

Private Sub Workbook_Open()
    Cells(Rows.Count, 1).End(xlUp).Offset(1).Value = Now
    MsgBox "打開[ " & Me.Name & " ]活頁簿!!"
End Sub

145 關閉活頁簿時自動執行的程序¶

VERSION 1.0 CLASS
BEGIN
  MultiUse = -1  'True
END
Attribute VB_Name = "ThisWorkbook"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = True

Private Sub Workbook_BeforeClose(Cancel As Boolean)
    Dim myRange As Range, myflag As Boolean
    myflag = False
    For Each myRange In Range("A7:C7")
        If myRange.Value = "" Then
           myflag = True
        End If
    Next
    If myflag Then
       MsgBox "有漏記!!!"
       Cancel = True
    Else
       ActiveWorkbook.SaveAs Format(Date, "mmdd") & Range("C2").Value
    End If
End Sub
